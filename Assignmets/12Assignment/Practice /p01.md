class InvalidAgeException extends Exception {

        super(msg);
    }
}

public class Test {
    public static void main(String[] args) {
        try {
            throw new InvalidAgeException("Not eligible for voting");
        } catch (InvalidAgeException e) {
            System.out.println("Caught: " + e.getMessage());
        }
    }
}
